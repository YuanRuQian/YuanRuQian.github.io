const front = new Set([’(’,’{’,’[’])

let stack = [];
let line = '{[()]([])}()[]'.split('');
for(let letter of line) {
    let flag = true;
    if(front.has(letter)) {
        stack.push(letter);
    } else {
        let top = stack.pop();
        if(top!==letter) {
            flag = false;
            break;
        }
    }
}
(!flag || stack.length>0) ? print('No') : print('Yes');
